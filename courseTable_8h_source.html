9struct CourseTable {
10 u8* dlRomStart; // 0x00 ROM start for segment 6 DL data
11 u8* dlRomEnd; // 0x04 ROM end for segment 6 DL data
12 u8* vertexRomStart; // 0x08 ROM start for segment 4 vertex data
13 u8* vertexRomEnd; // 0x0C ROM end for segment 7?
14 u8* offsetRomStart; // 0x10 ROM start for uncompressed segment 9 texture and DL addresses
15 u8* offsetRomEnd; // 0x14 ROM end for uncompressed segment 9 texture and DL addresses
16 CourseVtx* vertexStart; // 0x18 segmented address of vertex data
17 u32 vertexCount; // 0x1C number of vertices in vertex data
18 u8* packedStart; // 0x20 packed display list start address
19 u8* finalDisplaylistOffset; // 0x24
20 u32* textures; // 0x20 segmented address of textures table
21 u16 unknown1; // 0x2C
22 u16 padding; // 0x2E
23};
